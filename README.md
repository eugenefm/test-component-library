## Install

```bash
npm install --save my-library

or

yarn add my-library
```

## Usage

```jsx
import React from 'react'

import { Demo } from 'my-library'

const Example = ({message}) => <Demo type={'warning'}>{message}</Demo>;

export default Example;
```
